# Q7 (用R实现)
nums<-rep(seq(1,30),10)
count<-0
while (TRUE){
  count<-count+1
  if (length(unique(nums))==15){
    break
  }
  if (count%%9==0){
    lab<-which(nums==nums[count])
    nums<-nums[-lab]
  }
}
print(unique(nums))
